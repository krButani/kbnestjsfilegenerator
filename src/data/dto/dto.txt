import { ApiProperty } from '@nestjs/swagger';
import { [ModuleName] } from './../entity/[SMPRO].entity';
import { DateService } from './../../../shared/utils/dates.service';

export class [ModuleName]Dto {
  @ApiProperty()
  id: string;

  @ApiProperty()
  readonly createdAt: string;

  @ApiProperty()
  readonly updatedAt: string;
  constructor(data: any) {
    data = data.dataValues ? data.dataValues : data;
    let dateService: DateService = new DateService();

    this.id = data.id;
    this.createdAt = dateService.convertDisplayDateAsMoment(
      data.createdAt,
      dateService.getDisplayOnlyDateFormat(),
    );
    this.updatedAt = dateService.convertDisplayDateAsMoment(
      data.updatedAt,
      dateService.getDisplayOnlyDateFormat(),
    );
  }
}
